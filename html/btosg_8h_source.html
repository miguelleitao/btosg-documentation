204 class btosgObject {
205  public:
206  // Main components
207  //osg::Geode *geo;
208  osg::ref_ptr<osg::PositionAttitudeTransform> model;
209  char *name;
210  btTransform init_state;
211  btRigidBody *body;
212  // other
213  btCollisionShape* shape;
214  float mass;
215  btosgObject() {
216  model = NULL;
217  body = NULL;
218  shape = NULL;
219  mass = 0.;
220  name = NULL;
221  init_state = btTransform();
222  };
223  virtual ~btosgObject() {
224  if (body) {
225  delete body->getMotionState();
226  delete body;
227  body = NULL;
228  }
229  if( model ) {
230  model->unref();
231  model = NULL;
232  }
233  if (shape) {
234  delete shape;
235  shape = NULL;
236  }
237  if (name) {
238  delete name;
239  name = NULL;
240  }
241  }
242  void setName(char const *n) {
244  name = strdup(n);
245  }
246  void setMass(double m) {
248  mass = m;
249  bool isDynamic = (mass != 0.f);
250 
251  btVector3 localInertia(0, 0, 0);
252  if (isDynamic && shape)
253  shape->calculateLocalInertia(mass, localInertia);
254  if ( body ) body->setMassProps(m, localInertia);
255  }
256  btosgVec3 getPosition() {
258  if (body) {
259  btTransform wTrans;
260  body->getMotionState()->getWorldTransform(wTrans);
261  return wTrans.getOrigin();
262  }
263  //if (model) return osg2bt_Vec3(model->getPosition());
264  if (model) return model->getPosition();
265  return btosgVec3(0.,0.,0.);
266  }
267  btosgQuat getRotation() {
269  if (body) {
270  btTransform wTrans;
271  body->getMotionState()->getWorldTransform(wTrans);
272  return wTrans.getRotation();
273  }
274  //if (model) return osg2bt_Quat(model->getAttitude());
275  if (model) return model->getAttitude();
276  return btosgQuat(1.,0.,0.,0.);
277  }
278  btosgVec3 getEuler() {
280  btosgQuat qt = getRotation();
281  return qt.toEuler();
282  }
283  void setPosition(const btosgVec3 &p) {
285  if (body) {
286  btTransform wTrans;
287  body->getMotionState()->getWorldTransform(wTrans);
288  wTrans.setOrigin(p);
289  body->setWorldTransform(wTrans);
290  body->getMotionState()->setWorldTransform(wTrans);
291  body->clearForces();
292  body->setLinearVelocity(btVector3(0,0,0));
293  body->setAngularVelocity(btVector3(0,0,0));
294  }
295  #ifdef AVOID_DIRECT_MODEL_UPDATE
296  else
297  #endif
298  { // Not required for dynamic objects.
299  #ifdef _DEBUG_
300  printf("set Position in non dynamic object\n");
301  #endif
302  if (model) {
303  model->setPosition(p);
304  }
305  }
306  }
307  void setPosition(float x, float y, float z) {
309  setPosition(btosgVec3(x,y,z));
310  }
311  void setRotation(btQuaternion q) {
313  if (body) {
314  btTransform wTrans;
315  body->getMotionState()->getWorldTransform(wTrans);
316  wTrans.setRotation(q);
317  //wTrans.setOrigin(btVector3(x,y,z));
318  body->setWorldTransform(wTrans);
319  body->getMotionState()->setWorldTransform(wTrans);
320  body->clearForces();
321  body->setLinearVelocity(btVector3(0,0,0));
322  body->setAngularVelocity(btVector3(0,0,0));
323  }
324  #ifdef AVOID_DIRECT_MODEL_UPDATE
325  else
326  #endif
327  { // Not required for dynamic objects.
328  #ifdef _DEBUG_
329  printf("setRotation in non dynamic object\n");
330  #endif
331  if (model) {
332  model->setAttitude(bt2osg_Quat(q));
333  }
334  }
335  }
336  void setRotation(float x, float y, float z, float w) {
338  setRotation(btQuaternion(x,y,z,w));
339  }
340 
341  void setRotation(osg::Quat q) {
343  setRotation(q[0],q[1],q[2],q[3]);
344  }

346  void setTexture(char const *fname); //< Sets a texture from an image file.
347  void setMaterial(osg::ref_ptr<osg::Material> mat) {
349  model->getOrCreateStateSet()->setAttributeAndModes(mat, osg::StateAttribute::ON);
350  }
351  void logPosition() {
353  btosgVec3 pos = getPosition();
354  if ( name )
355  std::cout << "Object " << name << " position " << pos[0] << " " << pos[1] << " " << pos[2] << std::endl;
356  else
357  std::cout << "Object _NO_NAME_ position " << pos[0] << " " << pos[1] << " " << pos[2] << std::endl;
358  }
359  virtual void update() {
363  if (body) {
364  btTransform wTrans;
365  body->getMotionState()->getWorldTransform(wTrans);
366  if ( model ) {
367  model->setAttitude(btosgQuat(wTrans.getRotation()));
368  model->setPosition(btosgVec3(wTrans.getOrigin()));
369  }
370  }
371  }
372  void reset() {
374  if ( body ) {
375  body->setWorldTransform(init_state);
376  body->getMotionState()->setWorldTransform(init_state);
377  body->clearForces();
378  if ( mass>0. ) {
379  body->setLinearVelocity(btVector3(0.,0.,0.));
380  body->setAngularVelocity(btVector3(0.,0.,0.));
381  body->activate(); // Required if the object was asleep
382  }
383  }
384  else { // Not required for dynamic objects.
385  if ( model ) {
386  model->setAttitude(btosgQuat(init_state.getRotation()));
387  model->setPosition(btosgVec3(init_state.getOrigin()));
388  }
389  }
390  }
391  void setInitState() {
394  if (body) body->getMotionState()->getWorldTransform(init_state);
395  }
396  void setInitState(btTransform iState) {
399  init_state = iState;
400  }
401  void createRigidBody() {
403  if ( ! shape ) {
404  fprintf(stderr,"Cannot create RigidBody without shape\n");
405  return;
406  }
407  btDefaultMotionState* mState = new
408  btDefaultMotionState(btTransform(btQuaternion(0,0,0,1),btVector3(0.,0.,0.)));
409  btVector3 inertia(0,0,0);
410  shape->calculateLocalInertia(mass,inertia);
411  btRigidBody::btRigidBodyConstructionInfo cInfo(mass,mState,shape,inertia);
412  //printf("mass: %f\n",mass);
413  cInfo.m_restitution = 0.9f;
414  cInfo.m_friction = 10.f;
415  body = new btRigidBody(cInfo);
416  if ( !body ) fprintf(stderr,"Error creating btBody\n");
417  }
418  void loadObjectModel(char const *fname);
419 };
