239 class btosgObject {
240 public:
241  // Main components
242  osg::ref_ptr<osg::PositionAttitudeTransform> model;
243  char *name;
244  btTransform init_state;
245  btRigidBody *body;
246  // other
247  btCollisionShape* shape;
248  float mass;
249  btosgObject() {
250  model = NULL;
251  body = NULL;
252  shape = NULL;
253  mass = 0.;
254  name = NULL;
255  init_state = btTransform();
256  };
257  virtual ~btosgObject() {
258  if (body) {
259  delete body->getMotionState();
260  delete body;
261  body = NULL;
262  }
263  if( model ) {
264  //model->unref();
265  //model = NULL;
266  }
267  if (shape) {
268  delete shape;
269  shape = NULL;
270  }
271  if (name) {
272  free(name);
273  name = NULL;
274  }
275  }
276  void print();
277  void setName(char const *n) {
279  name = strdup(n);
280  }
281  void setMass(double m) {
284  mass = m;
285  bool isDynamic = (mass != 0.f);
286 
287  btVector3 localInertia(0, 0, 0);
288  if (isDynamic && shape)
289  shape->calculateLocalInertia(mass, localInertia);
290  if ( body ) body->setMassProps(m, localInertia);
291  }
292  btosgVec3 getPosition() {
294  if (body) {
295  btTransform wTrans;
296  body->getMotionState()->getWorldTransform(wTrans);
297  return wTrans.getOrigin();
298  }
299  if (model) return model->getPosition();
300  return btosgVec3(0.,0.,0.);
301  }
302  btosgQuat getRotation() {
304  if (body) {
305  btTransform wTrans;
306  body->getMotionState()->getWorldTransform(wTrans);
307  return wTrans.getRotation();
308  }
309  if (model) return model->getAttitude();
310  return btosgQuat(0.,0.,0.,1.);
311  }
312  btosgVec3 getEuler() {
314  btosgQuat qt = getRotation();
315  return qt.toEuler();
316  }
317  void setTransform(const btTransform &wTrans) {
319  if (body) {
320  body->setWorldTransform(wTrans);
321  body->getMotionState()->setWorldTransform(wTrans);
322  body->clearForces();
323  body->setLinearVelocity(btVector3(0,0,0));
324  body->setAngularVelocity(btVector3(0,0,0));
325  }
326  }
327  void setPosition(const btosgVec3 &p) {
329  if (body) {
330  btTransform wTrans;
331  body->getMotionState()->getWorldTransform(wTrans);
332  wTrans.setOrigin(p);
333  setTransform(wTrans);
334  }
335  #ifdef AVOID_DIRECT_MODEL_UPDATE
336  else
337  #endif
338  { // Not required for dynamic objects.
339  #ifdef _DEBUG_
340  printf("set Position in non dynamic object\n");
341  #endif
342  if (model) {
343  model->setPosition(p);
344  }
345  }
346  }
347  void setPosition(float x, float y, float z) {
349  setPosition(btosgVec3(x,y,z));
350  }
351  void setRotation(btosgQuat q) {
353  if (body) {
354  btTransform wTrans;
355  body->getMotionState()->getWorldTransform(wTrans);
356  wTrans.setRotation(q);
357  body->setWorldTransform(wTrans);
358  body->getMotionState()->setWorldTransform(wTrans);
359  body->clearForces();
360  body->setLinearVelocity(btVector3(0,0,0));
361  body->setAngularVelocity(btVector3(0,0,0));
362  }
363  #ifdef AVOID_DIRECT_MODEL_UPDATE
364  else
365  #endif
366  { // Not required for dynamic objects.
367  #ifdef _DEBUG_
368  printf("setRotation in non dynamic object\n");
369  #endif
370  if (model) {
371  model->setAttitude(q);
372  }
373  }
374  }
375  void setRotation(float x, float y, float z, float w) {
377  setRotation(btosgQuat(x,y,z,w));
378  }
379 
380  void setTexture(char const *fname); //< Sets a texture from an image file.
381  void setMaterial(osg::ref_ptr<osg::Material> mat) {
383  model->getOrCreateStateSet()->setAttributeAndModes(mat, osg::StateAttribute::ON);
384  }
385  void logPosition() {
387  btosgVec3 pos = getPosition();
388  if ( name )
389  std::cout << "Object " << name << " position " << pos[0] << " " << pos[1] << " " << pos[2] << std::endl;
390  else
391  std::cout << "Object _NO_NAME_ position " << pos[0] << " " << pos[1] << " " << pos[2] << std::endl;
392  }
393  virtual void update() {
397  if (body) {
398  btTransform wTrans;
399  body->getMotionState()->getWorldTransform(wTrans);
400  if ( model ) {
401  model->setAttitude(btosgQuat(wTrans.getRotation()));
402  model->setPosition(btosgVec3(wTrans.getOrigin()));
403  }
404  }
405  }
406  void reset();
407  void setInitState() {
410  if (body) body->getMotionState()->getWorldTransform(init_state);
411  }
412  void setInitState(btTransform iState) {
415  init_state = iState;
416  }
417  void createRigidBody();
418  void loadObjectModel(char const *fname);
419 };
